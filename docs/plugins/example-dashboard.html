Example: Dashboard Handler

A dashboard handler adds a custom controls panel that appears inside a source row when the user expands it. This is useful when the built-in controls are not enough for a specific source.

Scenario

You have a browser source called _Scoreboard that takes a score URL parameter. Instead of editing the URL manually each time, you want +/- buttons directly in the dashboard row.

Code

// plugin.js
(function() {
  let score = 0;

  CustomHandlerPlugins.register({
    name: 'ScoreboardPlugin',
    isExternal: true,

    canHandle(sourceKind, sourceName) {
      return sourceName === '_Scoreboard';
    },

    async execute(options, sourceName, displayName) {
      const row = document.createElement('div');
      row.style.display = 'flex';
      row.style.gap = '8px';
      row.style.alignItems = 'center';

      const label = document.createElement('span');
      label.textContent = `Score: ${score}`;

      const minus = document.createElement('button');
      minus.textContent = '−';

      const plus = document.createElement('button');
      plus.textContent = '+';

      async function updateScore() {
        label.textContent = `Score: ${score}`;
        await window.PluginUtils.setBrowserSourceUrlWithParams(
          sourceName,
          await window.PluginUtils.getSourceURL(sourceName),
          { score: String(score) }
        );
      }

      minus.addEventListener('click', () => { score = Math.max(0, score - 1); updateScore(); });
      plus.addEventListener('click', () => { score++; updateScore(); });

      row.appendChild(minus);
      row.appendChild(label);
      row.appendChild(plus);
      options.appendChild(row);
    }
  });
})();

Result

Expand the _Scoreboard source row and you get + and − buttons. Each click updates the score query parameter in the browser source URL.
