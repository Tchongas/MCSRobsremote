Example: Dashboard Handler

A dashboard handler adds more controls to the sources options in the dashboard, it can be used to add custom controls to specific source types, like all texts, or more specific source like all browser sources that are youtube links.

Example

This plugin makes every Text source in the dashboard have a button to convert the text to lowercase.

Code

// plugin.js
(function () {
  CustomHandlerPlugins.register({
    name: 'TextFormatter',
    version: '1.0.0',
    canHandle: (kind) => kind === 'text_gdiplus_v3',
    async execute(options, sourceName, displayName) {
      const row = document.createElement('div');
      row.className = 'dash-option-row';
      const btn = document.createElement('button');
      btn.textContent = 'aa Lowercase';
      btn.addEventListener('click', async () => {
        btn.disabled = true;
        try {
          const current = await window.PluginUtils.getSourceText(sourceName);
          const lower = current.toLowerCase();
          if (lower === current) {
            window.uiHelpers?.logInfo('Already lowercase.', 'TextFormatter');
          } else {
            await window.PluginUtils.setTextSource(sourceName, lower);
            window.uiHelpers?.logSuccess(`"${displayName}" → lowercase`, 'TextFormatter');
            const ta = options.querySelector('textarea.text-input');
            if (ta) ta.value = lower;
          }
        } catch (err) {
          window.uiHelpers?.logError(String(err), 'TextFormatter');
        } finally {
          btn.disabled = false;
        }
      });
      row.appendChild(btn);
      options.appendChild(row);
    },
    priority: () => 0,
    cleanup() {}
  });
})();
