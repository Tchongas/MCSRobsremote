Example: Workspace Popup — OverlayEditor

A workspace popup opens a real Electron window with custom HTML. It can call back into the host plugin via RPC to read and write OBS data. This example is a popup that shows all text sources in the current scene as editable inputs, positioned where they appear on the OBS canvas.

How it works

  • The sidebar button reads the current scene name and opens a popup window.
  • The popup script calls getSources via RPC to fetch all text sources and their OBS transforms.
  • Each source is rendered as a transparent <input> positioned at its scaled OBS coordinates over a black canvas.
  • Editing a field and pressing Enter (or clicking away) calls setText via RPC to update the source in OBS. The border flashes green on success.

RPC Handlers

RPC handlers are registered in the host plugin with registerPopupRpcHandlers. They run in the main renderer context and have full access to PluginUtils and obsAPI. The popup calls them with window.pluginPopupHost.call(method, ...args).

Full Plugin Code

// plugin.js
(function () {
  const PLUGIN = 'OverlayEditor';
  const CANVAS_W = 960;
  const CANVAS_H = 540;
  const OBS_W = 1920;
  const OBS_H = 1080;

  CustomHandlerPlugins.register({
    name: PLUGIN, version: '1.0.0', isExternal: true,
    canHandle: () => false, execute: async () => {}
  });

  // RPC handlers — run in the main window, called from the popup
  window.PluginUtils.registerPopupRpcHandlers(PLUGIN, {
    async getSources(sceneName) {
      const items = await window.PluginUtils.listSceneItems(sceneName);
      const textItems = items.filter(i => i.inputKind === 'text_gdiplus_v3');
      const results = await Promise.all(textItems.map(async (item) => {
        try {
          const [trRes, text] = await Promise.all([
            window.obsAPI.sceneItems.getTransform(sceneName, item.sceneItemId),
            window.PluginUtils.getSourceText(item.sourceName)
          ]);
          const tr = trRes?.sceneItemTransform || trRes;
          return {
            name: item.sourceName,
            display: item.sourceName.replace(/^_/, ''),
            sceneItemId: item.sceneItemId,
            x: Number(tr?.positionX ?? 0), y: Number(tr?.positionY ?? 0),
            w: Number(tr?.width ?? 200), h: Number(tr?.height ?? 40),
            text
          };
        } catch (_) { return null; }
      }));
      return results.filter(Boolean);
    },
    async setText(name, text) {
      await window.PluginUtils.setTextSource(name, text);
    }
  });

  // Sidebar button — opens the popup window
  window.PluginUtils.registerSidebarButton(PLUGIN, 'overlay-editor-open', 'Overlay Editor', (() => {
    let busy = false;
    return async () => {
      if (busy) return;
      busy = true;
      const sceneName = document.getElementById('sceneSelect')?.value || '';
      try {
        await window.PluginUtils.openWorkspacePopup(PLUGIN, {
          title: 'Overlay Editor',
          width: CANVAS_W + 40,
          height: CANVAS_H + 60,
          // Minimal bodyHtml — window shows instantly, content loads via RPC
          bodyHtml: '<div id="st" style="padding:12px;color:#aaa;font-family:sans-serif">Loading...</div><div id="cv" style="position:relative;background:#000;display:none"></div>',
          // script runs inside the popup window
          script: `
            const host = window.pluginPopupHost;
            const scene = '/* scene name injected at open time */';
            const W = 960, H = 540, SX = W / 1920, SY = H / 1080;
            const st = document.getElementById('st');
            const cv = document.getElementById('cv');

            host.call('getSources', scene).then(srcs => {
              st.style.display = 'none';
              cv.style.display = 'block';
              cv.style.width = W + 'px';
              cv.style.height = H + 'px';

              srcs.forEach(s => {
                const wrap = document.createElement('div');
                wrap.style.cssText = 'position:absolute;display:flex;flex-direction:column';
                wrap.style.left = (s.x * SX) + 'px';
                wrap.style.top  = (s.y * SY) + 'px';
                wrap.style.width = Math.max(s.w * SX, 120) + 'px';

                const inp = document.createElement('input');
                inp.type = 'text';
                inp.value = s.text;
                inp.style.cssText = 'width:100%;font-size:16px;background:transparent;border:1px solid transparent;color:#eee;outline:none';

                inp.onfocus = () => inp.style.borderColor = '#555';
                inp.onblur = async () => {
                  try {
                    await host.call('setText', s.name, inp.value);
                    inp.style.borderColor = '#4a4';
                    setTimeout(() => inp.style.borderColor = 'transparent', 800);
                  } catch(e) { inp.style.borderColor = '#a44'; }
                };
                inp.onkeydown = e => { if (e.key === 'Enter') inp.blur(); };

                wrap.appendChild(inp);
                cv.appendChild(wrap);
              });

              if (!srcs.length) {
                st.style.display = 'block';
                st.textContent = 'No text sources found in this scene.';
              }
            }).catch(e => { st.textContent = 'Error: ' + e.message; });
          `
        });
      } catch(e) {
        window.uiHelpers?.logError('Failed to open: ' + e, PLUGIN);
      } finally {
        busy = false;
      }
    };
  })(), { icon: '🗺' });
})();
  • The popup is a separate process - it has no direct access to OBS, PluginUtils, or the DOM of the main window. Everything goes through RPC.
  • RPC handlers run in the main window - they have full access to PluginUtils, obsAPI, and the rest of the plugin environment. The popup only receives serializable return values (plain objects, arrays, strings, numbers).
  • The script string runs inside the popup - it is injected as a <script> tag. You can interpolate values from the host context at open time using template literals, which is how the scene name gets passed in.
  • the popup window is shown as soon as it renders. Put a loading state in bodyHtml and populate the real content once your first RPC call resolves.
