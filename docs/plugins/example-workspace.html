Example: Modal Workspace

Plugins can open a panel inside the ROBS window (modal) or a separate window (popup). This is for plugins that need more UI than a single button click.

Scenario

You want a plugin button that opens a panel with a form — for example, a list of runners where you can edit names and click "Sync" without going back to a file.

In-App Modal

// plugin.js
(function() {
  window.PluginUtils.registerModalSidebarButton(
    'RunnerEditor',
    'runner-editor-btn',
    'Edit Runners',
    async ({ mount, close }) => {
      // Build your UI inside `mount`
      const form = document.createElement('div');
      form.style.padding = '16px';

      const input = document.createElement('input');
      input.type = 'text';
      input.placeholder = 'Runner name';

      const saveBtn = document.createElement('button');
      saveBtn.textContent = 'Save';
      saveBtn.addEventListener('click', async () => {
        await window.PluginUtils.setTextSource('_Runner1Name', input.value);
        window.uiHelpers.logSuccess('Runner name updated', 'RunnerEditor');
        close();
      });

      form.appendChild(input);
      form.appendChild(saveBtn);
      mount.appendChild(form);
    },
    { title: 'Edit Runners' }
  );
})();

Separate Popup Window

For more complex UIs, open a real Electron window. The popup can call back into the host via RPC. 

// plugin.js
(function() {
  window.PluginUtils.registerSidebarButton(
    'RunnerEditor',
    'runner-editor-popup',
    'Editor Window',
    async () => {
      await window.PluginUtils.openWorkspacePopup('RunnerEditor', {
        title: 'Runner Editor',
        width: 800,
        height: 500,
        bodyHtml: '',
        script: `
          document.getElementById('save').addEventListener('click', async () => {
            const name = document.getElementById('nameInput').value;
            await popupHost.call('setName', name);
          });
        `,
        rpcHandlers: {
          async setName(name) {
            await window.PluginUtils.setTextSource('_Runner1Name', name);
          }
        }
      });
    }
  );
})();

Notes

  • In modals, build DOM directly into the mount element.
  • In popups, the script string runs inside the popup window. Use pluginPopupHost.call(method, ...args) to call the rpcHandlers defined in the host plugin.
  • Only one modal can be open at a time. Opening a second one closes the first.
