Example: Workspace

This example shows how to save and restore a workspace layout or state from a plugin.

Scenario

You run different event formats that need different source layouts. A plugin can save the current scene collection name and source visibility states, then restore them later.

Code

const { pluginutils } = require('../../pluginutils');
const fs = require('fs');
const path = require('path');

const workspaceFile = path.join(__dirname, 'workspace.json');

function saveWorkspace() {
  const state = {
    collection: pluginutils.getCurrentSceneCollection(),
    sources: pluginutils.getSources().filter(s => s.name.startsWith('_'))
  };
  fs.writeFileSync(workspaceFile, JSON.stringify(state, null, 2));
}

function loadWorkspace() {
  const state = JSON.parse(fs.readFileSync(workspaceFile, 'utf8'));
  pluginutils.setCurrentSceneCollection(state.collection);
  state.sources.forEach(s => pluginutils.setSourceVisibility(s.name, s.enabled));
}

module.exports = { saveWorkspace, loadWorkspace };

Result

Two buttons appear in the sidebar: one saves the current workspace, the other restores it. Store any extra metadata you need in the JSON file.
